pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                echo 'Executing Checkout'
                git branch: 'main', url: 'https://github.com/Veverita-Engineering/Customer-Portal'
            }
        }
        stage('Build') {
            steps {
                echo 'Executing Build'
                nodejs(nodeJSInstallationName: 'node-lts'){
                    sh 'node --version'
                    sh 'npm --version'
                    sh 'npm i'
                    sh 'npm run build'
                    sh 'npm test -- --reporters=jest-junit'
                }
            }
        }
        stage('Parallel'){
        failFast true
        parallel {
        stage('Test') {
            steps {
                echo 'Executing Test'
                nosejs(nodeJSInstallationName: 'node-lts'){
                    sh 'npm test -- --reporters=jest-junit'
                }
            }
        }
        post {
        always {
            junit: 'reports/junit.xml'
        }
    }
        stage('SCA') {
            steps {
                echo 'Executing SCA'
            }
        } 
    }
        }
        stage('Deploying in stage') {
            steps {
                echo 'Deploying in stage'
            }
        }
        stage('Deploying in prod') {
            steps {
                echo 'Deploying in prod'
            }
        }
        
    }
    post {
        success{
            archiveArtifacts artifacts: 'build/**', fingerprint:true
        }
    }
}
